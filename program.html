BRIGHTNESS Pada opsi ini akan mengatur tingkat kecerahan pada gambar yang kita panggil dengan simbol +/-

#Brightness
Image1 <- Image + 0.2
Image2 <- Image - 0.2
par(mfrow= c(1,2))
plot (Image1)
plot (Image2)

ADJUSTING CONTRAST diopsi ini gambar akan diubah kontrasnya hanya saja disini menggunakan simbol sama dengan (=) secara default gambar bernilai 1

#Adjusting Contrast
Image3 <- Image * 0.5
Image4 <- Image * 2
par (mfrow= c(1,2))
plot(Image3)
plot(Image4)

GAMMA CORRECTION pada opsi ini image akan dirubah gammanya untuk caranya sama seperti kontras namun untuk gamma menggunakan ^,

#Gamma Correction
Image5 <- Image ^ 2
Image6 <- Image ^ 0.7
par(mfrow= c(1,2))
plot(Image5)
plot(Image6)

CROPPING di opsi ini akan memotong gambar yang kita panggil. Dengan cara seperti di bawah ini

#cropping
display(Image[100:650, 150:200,])

SPATIAL TRANSFORMATION untuk spatial tranformation image akan di setting rotasinya. sesuai value yang ada di Imagetr <- translate(rotate(Image, 45), c(50, 0))

#Spatial Transformation
Imagetr <- translate(rotate(Image, 45), c(50,0))
display(Imagetr)

FILTERING untuk filtering merupakan cara yang di bilang paling panjang dan berat untuk di jalankan. namun dengan filter akan berdampak sangat besar pada gambar. untuk hasil dan caranya seperti berikut.

#filtering
flow <- makeBrush(21, shape= 'disc', step=FALSE)^2
flow <- flow/sum(flow)
Image.flow <- filter2(Image, flow)
display(Image.flow)

fHigh <- matrix(1, nc = 3, nr = 3)
fHigh[2, 2] <- -8
Image.fHigh <- filter2(Image, fHigh)
display(Image.fHigh)
